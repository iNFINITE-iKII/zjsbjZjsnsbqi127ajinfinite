local function AnimateColor(instance, property, color, time)
    local tweenInfo = TweenInfo.new(time or 0.3, Enum.EasingStyle.Quad, Enum.EasingDirection.Out)
    local tween = TweenService:Create(instance, tweenInfo, {[property] = color})
    tween:Play()
    return tween
end

-- Animasi Masuk

-- Logika Verifikasi Utama
local isChecking = false
CheckBtn.MouseButton1Click:Connect(function()
    if isChecking then return end
    local input = KeyInput.Text

    if input == "" then
        AnimateColor(InputStroke, "Color", Theme.Error, 0.2)
        task.wait(1)
        AnimateColor(InputStroke, "Color", Theme.Accent, 0.2)
        return
    end

    isChecking = true
    CheckBtn.Text = "Verifying with Server..."
    AnimateColor(CheckBtn, "BackgroundColor3", Theme.TopBar, 0.3)
    AnimateColor(CheckBtn, "TextColor3", Theme.TextSecondary, 0.3)

    -- 1. Mendapatkan HWID Pemain
    local playerHWID = "UNKNOWN"
    pcall(function()
        if gethwid then
            playerHWID = gethwid()
        else
            playerHWID = game:GetService("RbxAnalyticsService"):GetClientId()
        end
    end)

    -- 2. Membentuk URL lengkap dengan Key dan HWID
    local fetchURL = ServerURL .. input .. "&hwid=" .. playerHWID

    -- 3. Meminta validasi ke server API Replit/Railway
    local success, response = pcall(function()
        return game:HttpGet(fetchURL)
    end)

    if success then
        local data = HttpService:JSONDecode(response)

        if data.status == "success" then
            CheckBtn.Text = "Access Granted!"
            AnimateColor(CheckBtn, "BackgroundColor3", Theme.Success, 0.3)
            AnimateColor(CheckBtn, "TextColor3", Color3.fromRGB(20, 20, 20), 0.3)
            AnimateColor(MainStroke, "Color", Theme.Success, 0.5)

            task.wait(1)
            TweenService:Create(MainFrame, TweenInfo.new(0.3, Enum.EasingStyle.Quad, Enum.EasingDirection.In), {
                Size = UDim2.new(0, 0, 0, 0),
                Position = UDim2.new(0.5, 0, 0.5, 50)
            }):Play()
            task.wait(0.3)
            KeySystemUI:Destroy()
            
            -- ========================================================
            -- EKSEKUSI SCRIPT UTAMA SOUL-IRON DI SINI
            -- (Posisinya sekarang aman di dalam pengecekan "success")
            -- ========================================================
            print("Sistem Valid! Memuat Soul-Iron V1...")
            
            local loadSuccess, loadError = pcall(function()
                loadstring(game:HttpGet("https://raw.githubusercontent.com/iNFINITE-iKII/Soul-Iron/refs/heads/main/V1"))()
            end)
            
            if not loadSuccess then
                warn("Gagal memuat script utama: " .. tostring(loadError))
            end
            
        else
            CheckBtn.Text = data.message or "Invalid or Expired Key"
            AnimateColor(CheckBtn, "BackgroundColor3", Theme.Error, 0.3)
            AnimateColor(CheckBtn, "TextColor3", Theme.Text, 0.3)
            AnimateColor(MainStroke, "Color", Theme.Error, 0.3)

            task.wait(1.5)
            CheckBtn.Text = "Verify Access"
            AnimateColor(CheckBtn, "BackgroundColor3", Theme.Accent, 0.3)
            AnimateColor(CheckBtn, "TextColor3", Color3.fromRGB(20, 20, 20), 0.3)
            AnimateColor(MainStroke, "Color", Theme.Accent, 0.5)
        end
    else
        CheckBtn.Text = "Server Offline / Error!"
        AnimateColor(CheckBtn, "BackgroundColor3", Theme.Error, 0.3)
        AnimateColor(CheckBtn, "TextColor3", Theme.Text, 0.3)
        AnimateColor(MainStroke, "Color", Theme.Error, 0.3)
        
        task.wait(2)
        CheckBtn.Text = "Verify Access"
        AnimateColor(CheckBtn, "BackgroundColor3", Theme.Accent, 0.3)
        AnimateColor(CheckBtn, "TextColor3", Color3.fromRGB(20, 20, 20), 0.3)
        AnimateColor(MainStroke, "Color", Theme.Accent, 0.5)
    end
    
    isChecking = false
end)
